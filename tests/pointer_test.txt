typedef struct string {
    int   (*get_size)(struct string*);
    char* (*get_body)(struct string*);
    int   (*set_body)(struct string*, char*);
} string_t;

void foo() {
    string_t s;
    void* b = &s;
    ((string_t*)((int)b))->get_size();
    ((string_t*)b)->get_body();
    ((string_t*)b)->set_body("Hello world!");
}
